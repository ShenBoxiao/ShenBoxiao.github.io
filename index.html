# 我们先创建两个对象
Mr = Boy("B.X. Shen", date = 1996-10-09);
Miss = Girl("W.W. Yu", date = 1997-04-09);
# 在19年8月25号，他们在微信上第一次认识，是Miss先打的招呼哦
Miss.know(Mr, said = ‘Hello’, site = ‘Wechat’)
# 之后似乎聊得还不错，在19年9月11号，他们见面了(Miss真可爱)
Mr.meet(Miss, mood = happy)
# Mr很喜欢Miss，见面最后的小插曲也让Mr很心疼，他摊牌了(嘻嘻)
Mr.fell_in_love(Miss, mood = anxious & excited)
# 最终在19年9月15日，他们升华了友谊
Miss.accepted(Mr)
# 从此开始了异地恋之旅
Couple(Mr, Miss, relationship = long-distance)
# 相互想念让人痛苦，不过幸运的是Mr几乎每个月都可以去看望Miss几天
if Mr.see(Miss, frequency = 1/month):
Mr.with(Miss, mood = happy)
else:
Mr.mood = sad
# 他们一起出去玩
Play(Mr, Miss)
# 一起吃好吃的
Cuisine(Mr, Miss)
# 一起体验？？？
Experience(Mr, Miss, hand = True, hug = True, kiss = True, mood = warm && happy)
# 中间也会有一些感动，当然，也有一些不开心
Progress(Mr, Miss, moved = sometimes, conflicts = sometimes)
# 但是最后还是能一起愉快的玩耍(哈哈)
Delight_together(Mr, Miss, conflicts = MM.close)
# 遇见喜欢的人多不容易啊，真希望Mr和Miss永远在一起(嘿嘿)
Mr.live_together_with(Miss, time = forever, hope = True)

# 永远爱你，宝贝
